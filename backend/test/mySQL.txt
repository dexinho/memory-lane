CREATE TABLE users (
    users_id INT PRIMARY KEY AUTO_INCREMENT UNIQUE,
    email VARCHAR(30) UNIQUE NOT NULL,
    password VARCHAR(40) NOT NULL
);

CREATE TABLE users_info (
    user_id INT PRIMARY KEY,
    first_name VARCHAR(30) NOT NULL,
    last_name VARCHAR(30) NOT NULL,
    age INT,
    birth_date DATE,
    country VARCHAR(20),
    FOREIGN KEY (user_id)
        REFERENCES users (id)
        ON DELETE CASCADE
);

CREATE TABLE pictures (
    picture_id INT PRIMARY KEY AUTO_INCREMENT UNIQUE,
    picture_url VARCHAR(255) UNIQUE
);

CREATE TABLE user_pictures (
    user_id INT,
    pictures_id INT,
    PRIMARY KEY (user_id , pictures_id),
    FOREIGN KEY (user_id)
        REFERENCES user_details (user_id)
        ON DELETE CASCADE,
    FOREIGN KEY (pictures_id)
        REFERENCES pictures (id)
        ON DELETE CASCADE
);
